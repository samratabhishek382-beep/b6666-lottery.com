<script>
let balance = 1000;
let betChoice = null;
let betAmount = 100;
let timeLeft = 60;

function updateBalance() {
document.getElementById("balance").innerText = balance;
}

function deposit() {
balance += 500;
updateBalance();
alert("Demo Deposit: 500 Coins Added");
}

function withdraw() {
alert("Demo Withdraw! (No real money)");
}

function bet(choice) {
if (betChoice) {
alert("Already placed a bet!");
return;
}
if (balance < betAmount) {
alert("Not enough balance!");
return;
}
balance -= betAmount;
updateBalance();
betChoice = choice;
document.getElementById("result").innerText = "You bet on: " + choice;
}

function roundResult() {
let colors = ["Green","Red","Violet"];
let number = Math.floor(Math.random() * 10);
let outcomeColor = colors[Math.floor(Math.random() * colors.length)];
let resultText = `Result: ${number} (${outcomeColor})`;

if (betChoice !== null) {
if (betChoice === outcomeColor) {
let win = betAmount * (betChoice === "Violet" ? 4.5 : 2);
balance += win;
resultText += ` üéâ You Win ${win} coins!`;
} else if (betChoice === number) {
let win = betAmount * 9;
balance += win;
resultText += ` üéâ You Win ${win} coins!`;
} else {
resultText += ` ‚ùå You Lost!`;
}
betChoice = null;
}

updateBalance();
document.getElementById("result").innerText = resultText;
timeLeft = 60;
}

setInterval(() => {
if (timeLeft > 0) {
timeLeft--;
document.getElementById("time").innerText = timeLeft;
} else {
roundResult();
}
}, 1000);

updateBalance();
</script>
